# 选择排序

```js
//选择排序
function selsetSort(arr) {
  var len = arr.length;
  var index;
  for (var i = 0; i < len - 1; i++) {
    index = i;
    for (var j = i + 1; j < len; j++) {
      if (arr[index] > arr[j]) {
        //寻找最小值
        index = j; //保存最小值的索引
      }
    }
    if (index != i) {
      [arr[i], arr[index]] = [arr[index], arr[i]];
    }
  }
  return arr;
}
```
